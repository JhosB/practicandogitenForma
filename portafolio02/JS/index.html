            class Persona {
                constructor(nombre, apellido, numDoc) {
                    this._nombre = nombre;
                    this._apellido = apellido;
                    this._numDoc = numDoc;
                }
            
                get nombre() {
                    return this._nombre;
                }
            
                set nombre(nombre) {
                    this._nombre = nombre;
                }
            
                get apellido() {
                    return this._apellido;
                }
            
                set apellido(apellido) {
                    this._apellido = apellido;
                }
            
                get numDoc() {
                    return this._numDoc;
                }
            
                set numDoc(numDoc) {
                    this._numDoc = numDoc;
                }
            }
